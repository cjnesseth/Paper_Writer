% ---------- 2. PJM in the industry ----------
\begin{frame}{PJM at a Glance}
  \begin{itemize}
    \item Regional transmission organization covering 13 states plus DC,
          about 65 million customers and 180 GW of capacity
    \item Operates three wholesale markets
      \begin{itemize}
        \item Energy pays for electricity produced
        \item Ancillary services pay for reserves and grid support
        \item Capacity pays for availability three years ahead
      \end{itemize}
    \item This paper studies the capacity market (annual BRA under RPM)
  \end{itemize}
\end{frame}

% ---------- 3. The $21B puzzle ----------
\begin{frame}{The \$21 Billion Question}
  \begin{itemize}
    \item In December 2024, Pennsylvania filed a FERC complaint after the
          2025/26 auction cleared at \$269.92/MW-day, an 800\% jump
    \item In January 2025, the Shapiro-PJM settlement capped the 2026/27
          and 2027/28 auctions at \$325/MW-day with a \$175/MW-day floor
    \item Settlement claims over \$21 billion in two-year
          consumer savings
  \end{itemize}
\end{frame}

% ---------- 4. Research Question ----------
\begin{frame}{Research Question}
  \vfill
  \begin{center}
    \Large
    What would PJM's 2026/27 and 2027/28 auctions have cleared at
    \textbf{absent the settlement}? \\[1em]
    And what is the actual revenue transfer it produces?
  \end{center}
  \vfill
\end{frame}

% ---------- 7. Model setup ----------
\begin{frame}{Model: Supply Function Equilibrium}
  $K$ symmetric sellers bid supply functions against the VRR demand,
  in the tradition of Klemperer--Meyer (1989)
  \vfill
  \begin{equation*}
    s'(p) \;=\; \frac{D'(p) \,+\, s(p)/(p - c)}{K - 1}
  \end{equation*}
  \vfill
  \begin{itemize}
    \item Holmberg (2008) boundary $s(\bar{p}) = \bar{q}$ selects the equilibrium
    \item Competitive fringe offers a block $Q_f$ at cost $c$
    \item $p^*$ solves $K \cdot s(p^*) + Q_f = D(p^*)$
  \end{itemize}
\end{frame}

% ---------- 8. At-cap is mechanical ----------
\begin{frame}{The At-Cap Regime Is Mechanical}
  \begin{itemize}
    \item When $K \leq 3$, the interior ODE cannot clear demand below $\bar{p}$
    \item The market then clears at $\bar{p}$ \emph{by construction}
          via the boundary condition
    \item The \$329 and \$333 figures come from PJM's VRR Point~(a),
          not from an independent price forecast
  \end{itemize}
  \vfill
  \begin{center}
    \alert{The substantive content is the \textit{regime}
    (at-cap vs.\ interior) and its dependence on $K$}
  \end{center}
\end{frame}

% ---------- 13. Limitations ----------
\begin{frame}{Limitations}
  \begin{itemize}
    \item New-design years cleared short of reliability req.: at-cap is
          overdetermined (strategic mechanism vs.\ physical scarcity)
    \item TPS mitigation not endogenous, may be the true binding constraint
    \item Firms assumed symmetric (cost heterogeneity bracketed)
    \item Holmberg boundary used as selection rule, not derived
    \item Static model - no retirement or entry response
  \end{itemize}
\end{frame}

% ---------- 14. Implications & Conclusion ----------
\begin{frame}{What the Settlement Does, and What It Does Not}
  \begin{itemize}
    \item Cap \$325 sits 53\% above Net CONE (\$212) - entry remains
          viable, so the concern is informational not cost recovery
    \item Structural channels (transmission, steeper VRR, entry) reduce
          markups without truncating the signal
    \item Temporary suppression is not a substitute for structural reform
  \end{itemize}
